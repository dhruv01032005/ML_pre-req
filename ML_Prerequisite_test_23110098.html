import time
import numpy as np

def matrix_multiplication(matrix1, matrix2):
    c1 = len(matrix1[0])
    r2 = len(matrix2)
    if (c1 != r2):
        return "INVALID INPUT"
    n = len(matrix1)
    k = len(matrix1[0])
    m = len(matrix2[0])
    ans = []
    for i in range(n):
        temp = []
        for j in range(m):
            temp.append(0);
        ans.append(temp)

    for i in range(n):
        for j in range(m):
            for l in range(k):
                ans[i][j] += matrix1[i][l] * matrix2[l][j]
    return ans

import numpy as np

def compute_argmin(arr):
    min_index = 0
    min_value = arr[0]

    for i in range(1, len(arr)):
        if arr[i] < min_value:
            min_value = arr[i]
            min_index = i

    return min_index
